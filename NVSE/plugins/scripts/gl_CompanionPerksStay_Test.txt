; Test script for Companion Perks Stay
; Force-hires and teleports all companions to the player.
; Enable via CompanionPerksStay.ini [Testing] bTestMode = 1
; Warning: This script recruits all companions. Due to engine limitations, some companions will be automatically dismissed to stay within the companion limit. Be aware that speaking to certain companions may trigger their associated quests. For example, talking to Christine will begin her Dead Money storyline.

float fTest
set fTest to GetINIFloat "Testing:bTestMode" "CompanionPerksStay.ini"

    return
